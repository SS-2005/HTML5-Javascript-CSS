<h1>Switch Case Statement</h1>
<script type="text/javascript">
var color=parseInt(prompt("Enter colour code 0-7"));
switch(color)
{
case 1:
alert("violet");
break;
case 2:
alert("indigo");
break;
case 3:
alert("blue");
break;
case 4:
alert("green");
break;
case 5:
alert("yellow");
break;
case 6:
alert("orange");
break;
case 7:
alert("red");
break;
default:
alert("Invalid Color");
break;
}
</script>
